var showLogging = true;
var block_size = 25;
var player = null;

function start() {
    
    var wall = imageLoadHelper("https://raw.githubusercontent.com/r0ckl33/PythonPushBox/master/assets/block.png", block_size, block_size);
    wall.setPosition(0, 0);
    add(wall);
    
    var egg = imageLoadHelper("https://raw.githubusercontent.com/r0ckl33/PythonPushBox/master/assets/egg.png", block_size, block_size);
    egg.setPosition(25, 0);
    add(egg);
    
    var floor = imageLoadHelper("https://raw.githubusercontent.com/r0ckl33/PythonPushBox/master/assets/floor.png", block_size, block_size);
    floor.setPosition(50, 0);
    add(floor);
    
    var nest = imageLoadHelper("https://raw.githubusercontent.com/r0ckl33/PythonPushBox/master/assets/nest.png", block_size, block_size);
    nest.setPosition(75, 0);
    add(nest);
    
    player = imageLoadHelper("https://raw.githubusercontent.com/r0ckl33/PythonPushBox/master/assets/player.png", block_size, block_size);
    add(player);
    
    keyDownMethod(keyDown);
    
    log("loaded");
}

function imageLoadHelper(imageUrl, w, h) {
    var image = new WebImage(imageUrl);
    image.setSize(w, h);
    image.setPosition(getWidth() / 2, getHeight() / 2);
    return image;
}

function log(message) {
    if (showLogging) {
        print(message);
    }
}

function keyDown(e) {
    if (e.keyCode == Keyboard.LEFT) {
		player.move(-block_size, 0);
	} else if (e.keyCode == Keyboard.RIGHT) {
		player.move(block_size, 0);
	} else if (e.keyCode == Keyboard.UP) {
		player.move(0, -block_size);
	} else if (e.keyCode == Keyboard.DOWN) {
		player.move(0, block_size);
	}
}
